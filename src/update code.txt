public static Date getCurrentDateTimeInVietnam() {
        // Lấy múi giờ của Việt Nam
        ZoneId vietnamZone = ZoneId.of("Asia/Ho_Chi_Minh");

        // Lấy ngày giờ hiện tại ở Việt Nam
        LocalDateTime currentDateTime = LocalDateTime.now(vietnamZone);

        // Chuyển đổi LocalDateTime thành java.util.Date
        Date currentDate = Date.from(currentDateTime.atZone(vietnamZone).toInstant());

        return currentDate;
    }
