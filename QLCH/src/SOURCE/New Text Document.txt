public class NumberToWords {
    private static final String[] units = {"", "nghìn", "triệu", "tỷ"};
    private static final String[] digits = {"", "một", "hai", "ba", "bốn", "năm", "sáu", "bảy", "tám", "chín"};

    public static void main(String[] args) {
        int number = 4318600;
        String words = convertNumberToWords(number);
        System.out.println(number + " đồng thành chữ: " + words);
    }

    public static String convertNumberToWords(int number) {
        if (number == 0) {
            return "Không đồng";
        }

        String words = "";
        int unitIndex = 0;

        do {
            int threeDigits = number % 1000;
            if (threeDigits != 0) {
                String part = convertThreeDigitsToWords(threeDigits);
                words = part + " " + units[unitIndex] + " " + words;
            }
            unitIndex++;
            number /= 1000;
        } while (number > 0);

        return words.trim() + " đồng";
    }

    private static String convertThreeDigitsToWords(int number) {
        String words = "";

        int hundredDigit = number / 100;
        int remainingDigits = number % 100;

        if (hundredDigit > 0) {
            words += digits[hundredDigit] + " trăm";
            if (remainingDigits > 0) {
                words += " ";
            }
        }

        if (remainingDigits > 0) {
            if (remainingDigits < 10) {
                words += digits[remainingDigits];
            } else if (remainingDigits < 20) {
                words += digits[remainingDigits - 10] + " mười";
            } else {
                int tenDigit = remainingDigits / 10;
                int unitDigit = remainingDigits % 10;

                words += digits[tenDigit] + " mươi";

                if (unitDigit > 0) {
                    words += " " + digits[unitDigit];
                }
            }
        }

        return words;
    }
}
